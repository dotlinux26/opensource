// Update the chat input event listener to allow spaces in messages document.getElementById('chatInput').addEventListener('keydown', function(event) { // Prevent default action if the input is empty and the key is being pressed if (event.key === 'Enter' && this.value.trim() !== '') { event.preventDefault(); sendMessage(this.value); this.value = ''; // Clear the input after sending } // Allow all keys including spaces to be input });
